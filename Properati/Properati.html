Muchas de estas publicaciones tienen sus campos surface nulos, los cuales son los m2 de la propiedad pero en sus descripciones esta referenciado, con esta funcion selecciono las propiedades que tienen los campos surface nulos pero que en su descripción contienen la siguiente estructura en orden
[0-9]* : numeros del 0 al 9 repetidos n veces
(,|\.| ) : “,”, “.” ó " "
([0-9]*( |)) : numeros del 0 al 9 repetidos n veces y el espacio " " o vacio ""
(mts2|m2|Mts2|mt2) : cualquiera de estos textos de unidades

propiedadestm <- propiedades %>%
   filter((is.na(surface_total) &
              is.na(surface_covered)) & !is.na(description)) %>%
   mutate(
      description = str_remove_all(description, "\n"),
      descriptionregex = str_extract(description, "[0-9]*(,|\\.| )([0-9]*( |))(mts2|m2|Mts2|mt2)")
   ) %>%
   filter(!is.na(descriptionregex))

Hago un pequeño filtro, luego agrego estos datos al campo surface_total y borro el campo descriptionregex

#Filtro
propiedadestm <- propiedadestm %>%
   mutate(
      descriptionregex = str_remove_all(descriptionregex, ",[0-9]*"),
      descriptionregex = str_remove_all(descriptionregex, "\\."),
      descriptionregex = str_remove_all(descriptionregex, "(mts2|m2|Mts2|mt2)"),
      descriptionregex = as.double(descriptionregex)
   ) %>%
   filter(!is.na(descriptionregex) &
             (descriptionregex > 15 & descriptionregex < 1000))

#Borrado de campo
propiedadestm <- propiedadestm %>%
   mutate(surface_total = descriptionregex) %>%
   select(-descriptionregex)

Creo la columna precio por m2

propiedades <- propiedades %>%
   mutate(pricem2 = if_else(is.na(surface_covered), price/surface_total, price/surface_covered))
means <- propiedades %>% 
   filter(!is.na(pricem2)) %>%
   group_by(city, property_type, operation_type) %>%
   summarise(means = mean(pricem2))
## `summarise()` regrouping output by 'city', 'property_type' (override with `.groups` argument)
